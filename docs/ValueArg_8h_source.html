271 bool ValueArg<T>::processArg(int *i, std::vector<std::string> &args) {
272  if (_ignoreable && Arg::ignoreRest())
273  return false;
274 
275  if (_hasBlanks(args[*i]))
276  return false;
277 
278  std::string flag = args[*i];
279 
280  std::string value = "";
281  trimFlag(flag, value);
282 
283  if (argMatches(flag)) {
284  if (_alreadySet) {
285  if (_xorSet)
286  throw (CmdLineParseException("Mutually exclusive argument already set!",
287  toString()));
288  else
289  throw (CmdLineParseException("Argument already set!", toString()));
290  }
291 
292  if (Arg::delimiter() != ' ' && value == "")
293  throw (ArgParseException("Couldn't find delimiter for this argument!",
294  toString()));
295 
296  if (value == "") {
297  (*i)++;
298  if (static_cast<unsigned int>(*i) < args.size())
299  _extractValue(args[*i]);
300  else
301  throw (ArgParseException("Missing a value for this argument!",
302  toString()));
303  } else
304  _extractValue(value);
305 
306  _alreadySet = true;
307  _checkWithVisitor();
308  return true;
309  } else
310  return false;
311 }
