403 inline void CmdLine::parse(std::vector<std::string> &args) {
404  bool shouldExit = false;
405  int estat = 0;
406 
407  try {
408  _progName = args.front();
409  args.erase(args.begin());
410 
411  int requiredCount = 0;
412 
413  for (int i = 0; static_cast<unsigned int>(i) < args.size(); i++) {
414  bool matched = false;
415  for (ArgListIterator it = _argList.begin(); it != _argList.end(); it++) {
416  if ((*it)->processArg(&i, args)) {
417  requiredCount += _xorHandler.check(*it);
418  matched = true;
419  break;
420  }
421  }
422 
423  // checks to see if the argument is an empty combined
424  // switch and if so, then we've actually matched it
425  if (!matched && _emptyCombined(args[i]))
426  matched = true;
427 
428  if (!matched && !Arg::ignoreRest())
429  throw (CmdLineParseException("Couldn't find match "
430  "for argument",
431  args[i]));
432  }
433 
434  if (requiredCount < _numRequired)
435  missingArgsException();
436 
437  if (requiredCount > _numRequired)
438  throw (CmdLineParseException("Too many arguments!"));
439 
440  } catch (ArgException &e) {
441  // If we're not handling the exceptions, rethrow.
442  if (!_handleExceptions) {
443  throw;
444  }
445 
446  try {
447  _output->failure(*this, e);
448  } catch (ExitException &ee) {
449  estat = ee.getExitStatus();
450  shouldExit = true;
451  }
452  } catch (ExitException &ee) {
453  // If we're not handling the exceptions, rethrow.
454  if (!_handleExceptions) {
455  throw;
456  }
457 
458  estat = ee.getExitStatus();
459  shouldExit = true;
460  }
461 
462  if (shouldExit)
463  exit(estat);
464 }

466 inline bool CmdLine::_emptyCombined(const std::string &s) {
467  if (s.length() > 0 && s[0] != Arg::flagStartChar())
468  return false;
469 
470  for (int i = 1; static_cast<unsigned int>(i) < s.length(); i++)
471  if (s[i] != Arg::blankChar())
472  return false;
473 
474  return true;
475 }

477 inline void CmdLine::missingArgsException() {
478  int count = 0;
479 
480  std::string missingArgList;
481  for (ArgListIterator it = _argList.begin(); it != _argList.end(); it++) {
482  if ((*it)->isRequired() && !(*it)->isSet()) {
483  missingArgList += (*it)->getName();
484  missingArgList += ", ";
485  count++;
486  }
487  }
488  missingArgList = missingArgList.substr(0, missingArgList.length() - 2);
489 
490  std::string msg;
491  if (count > 1)
492  msg = "Required arguments missing: ";
493  else
494  msg = "Required argument missing: ";
495 
496  msg += missingArgList;
497 
498  throw (CmdLineParseException(msg));
499 }
